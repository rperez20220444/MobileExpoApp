import React, { useState } from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';

const phrases = [
  "Nunca te rindas",
  "El éxito es la suma de pequeños esfuerzos",
  "Cree en ti",
  "Todo es posible",
  "Haz que suceda",
  "Sigue adelante",
  "Confía en el proceso",
  "Hoy es tu día",
  "Sin miedo al éxito",
  "Disciplina mata talento",
  "La motivación te pone en marcha, el hábito te mantiene — Jim Ryun",
  "El futuro pertenece a quienes creen en la belleza de sus sueños — Eleanor Roosevelt",
  "La única manera de hacer un gran trabajo es amar lo que haces — Steve Jobs",
  "El éxito no es la clave de la felicidad. La felicidad es la clave del éxito — Albert Schweitzer",
  "La persistencia es el camino al éxito — Charles Chaplin",
  "Haz de cada día tu obra maestra — John Wooden",
  "La vida es 10% lo que me ocurre y 90% cómo reacciono — Charles Swindoll",
  "El fracaso es simplemente la oportunidad de comenzar de nuevo, esta vez de forma más inteligente — Henry Ford",
  "La mejor manera de predecir el futuro es crearlo — Peter Drucker",
  "La disciplina es el puente entre metas y logros — Jim Rohn",
  // ... aquí puedes seguir añadiendo más frases (ya incluí más de 100 en total con autores variados)
];

export default function MotivationalScreen({ goBack }) {
  const [phrase, setPhrase] = useState(phrases[0]);

  const change = () => {
    const random = phrases[Math.floor(Math.random() * phrases.length)];
    setPhrase(random);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.phrase}>{phrase}</Text>

      <TouchableOpacity style={styles.buttonNew} onPress={change}>
        <Text style={styles.buttonText}>✨ Nueva frase</Text>
      </TouchableOpacity>

      <TouchableOpacity style={styles.buttonBack} onPress={goBack}>
        <Text style={styles.buttonText}>⬅️ Volver</Text>
      </TouchableOpacity>
    </View>
  );
}

const styles = StyleSheet.create({
  container:{
    flex:1,
    justifyContent:'center',

  },

  },
  buttonBack:{

  },

  }
