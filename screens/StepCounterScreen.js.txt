import React, { useState } from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';

export default function StepCounterScreen({ goBack }) {
  const [steps, setSteps] = useState(0);

  return (
    <View style={styles.container}>
      <Text style={styles.steps}>{steps}</Text>

      <TouchableOpacity style={styles.add} onPress={()=>setSteps(steps+1)}>
        <Text style={styles.text}>+1 Paso</Text>
      </TouchableOpacity>

      <TouchableOpacity style={styles.reset} onPress={()=>setSteps(0)}>
        <Text style={styles.text}>Reiniciar</Text>
      </TouchableOpacity>

      <Text style={styles.back} onPress={goBack}>← Volver</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container:{ flex:1, justifyContent:'center', alignItems:'center', backgroundColor:"#F5F7FA" },
  steps:{ fontSize:50, marginBottom:20 },
  add:{
    backgroundColor:"#4A90E2",
    padding:15,
    borderRadius:10,
    marginBottom:10
  },
  reset:{
    backgroundColor:"#FF5A5F",
    padding:15,
    borderRadius:10
  },
  text:{ color:"#FFF" },
  back:{ marginTop:20 }
});
