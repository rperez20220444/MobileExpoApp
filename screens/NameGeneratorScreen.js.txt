import React, { useState } from 'react';
import { View, Text, Button, FlatList, ScrollView } from 'react-native';

// Lista combinada de nombres femeninos y masculinos
const names = [
  "Sofía","Valentina","Isabella","Camila","Mariana","Lucía","Victoria","Gabriela","Daniela","Martina",
  "Mateo","Santiago","Matías","Sebastián","Alejandro","Samuel","Benjamín","Daniel","Lucas","Joaquín",
  // ... aquí continúan todos los nombres que compartiste (puedes pegarlos completos)
];

export default function NameGeneratorScreen({ goBack }) {
  const [current, setCurrent] = useState('');
  const [history, setHistory] = useState([]);

  const generate = () => {
    const random = names[Math.floor(Math.random() * names.length)];
    setCurrent(random);
    setHistory([random, ...history]);
  };

  const resetAndGoBack = () => {
    setCurrent('');
    setHistory([]);
    goBack();
  };

  return (
    <View style={{ padding: 20, flex: 1 }}>
      <Text style={{ fontSize: 24, marginBottom: 10 }}>{current}</Text>

      <Button title="Generar Nombre" onPress={generate} />

      <FlatList
        data={history}
        renderItem={({ item }) => <Text style={{ fontSize: 18 }}>{item}</Text>}
        keyExtractor={(item, index) => index.toString()}
        style={{ marginVertical: 20 }}
      />

      <Button title="Volver" onPress={resetAndGoBack} />
    </View>
  );
}
