import React, { useState } from 'react';
import { View, Text, TouchableOpacity, FlatList, StyleSheet } from 'react-native';

const names = ['Alpha','Nova','Shadow','Blaze','Orion','Titan','Luna'];

export default function NameGeneratorScreen({ goBack }) {
  const [current, setCurrent] = useState('');
  const [history, setHistory] = useState([]);

  const generate = () => {
    const random = names[Math.floor(Math.random()*names.length)];
    setCurrent(random);
    setHistory([random, ...history]);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>{current || "Presiona generar"}</Text>

      <TouchableOpacity style={styles.button} onPress={generate}>
        <Text style={styles.buttonText}>Generar</Text>
      </TouchableOpacity>

      <FlatList
        data={history}
        renderItem={({item}) => <Text style={styles.item}>{item}</Text>}
        keyExtractor={(item,i)=>i.toString()}
      />

      <Text style={styles.back} onPress={goBack}>← Volver</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container:{ flex:1, padding:20, backgroundColor:"#F5F7FA" },
  title:{ fontSize:26, textAlign:'center', marginBottom:20 },
  button:{
    backgroundColor:"#50E3C2",
    padding:15,
    borderRadius:10,
    marginBottom:20
  },
  buttonText:{ textAlign:'center', color:"#FFF" },
  item:{
    backgroundColor:"#FFF",
    padding:10,
    marginVertical:5,
    borderRadius:8
  },
  back:{ textAlign:'center', marginTop:10 }
});
